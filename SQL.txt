1. SELECT * FROM Контрагент

2.select
  Тип документа.Наименование as Тип договора
  Договор.Номер as № договора
  Договор.Дата_подписания as Дата подписания
  Поставщик.Наименование as Поставщик
  Плательщик.Наименование as Плательщик
from Договор
  inner join Тип_догумента
    on Тип_документа.ID_типа_документа = Договор.IDтипа_документа
  and Тип_документа.ID_типа_операции = Договор.IDтипа_операции
  inner join Контрагент as Поставщик
on Поставщик.ID_Контрагента = Договор.ID_Поставщика
  inner join Контрагент as Полательщик
on Полательщик.ID_Контрагента = Договор.ID_Плательщика

3.select
  Тип_документа.Наименование as Тип_документа
  Документ.Номер as № документа
  Договор.Дата_подписания as Дата подписания
  Позиции_документа.цена as Сумма по документу
  Договор.Номер as № договора
  Поставщик.Наименование as Поставщик
  Плательщик.Наименование as Плательщик
from Документ
  inner join Тип_догумента
    on Тип_документа.ID_типа_документа = Документ.ID_типа_документа
  and Тип_документа.ID_типа_операции = Документ.ID_типа_операции
  inner join Позиции_документа ON Позиции_документа.ID_документа = Документ.ID_документа
  inner join Контрагент as Поставщик
on Поставщик.ID_Контрагента = Договор.ID_Поставщика
  inner join Контрагент as Полательщик
on Полательщик.ID_Контрагента = Договор.ID_Плательщика
inner join Тип_операции ON Тип_догумента.ID_типа_операции = Тип_догумента.ID_типа_операции
WHERE Тип_операции.Наименование = "Расходные документы"
